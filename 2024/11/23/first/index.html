01 背包

#include <iostream>

using namespace std;

const int N = 1010;

int v[N],w[N];

int f[N];

int n,m;

int main()
{
    scanf("%d%d",&n,&m);
    for (int i = 1;i <= n;i ++) scanf("%d%d",&v[i],&w[i]);
    for (int i = 1;i <= n;i ++)
        for (int j = m;j >= v[i];j --)
            f[j] = max(f[j],f[j - v[i]] + w[i]);
            //在选v[i]的前提下继承上一层与剩余体积相同的最优解.
            //优化前当j<v[i]时数组对应位置为空,倒序节省空间复杂度.
    printf("%d",f[m]);
    return 0;
}
