#include <algorithm>
#include <iostream>
#include <cstring>
#include <vector>
#define int long long

using namespace std;

const int N = 100010;

int e[N],ne[N],h[N],idx;

int n,m,ans;

int ar[N],p[N],s[N];

vector<pair<int,int>> ur;

void add(int a,int b) {e[idx] = b,ne[idx] = h[a],h[a] = idx ++;}

int find(int x) {if (p[x] != x) p[x] = find(p[x]);return p[x];}

bool cmp(pair<int,int> a,pair<int,int> b) {return a.second > b.second;}

signed main()
{
    cin >> n >> m;

    for (int i =1 ;i<= n;i ++) p[i] = i,cin >> s[i];
    while (m --)
    {
        int a,b;
        cin >> a >> b;
        if (find(a) != find(b))
        {
            a = find(a),b = find(b);
            s[b] = max(s[b],s[a]);
            p[a] = b;
        }
    }
    for (int i = 1;i <= n;i ++)
    {
        if (p[i] == i)
        {
            ur.push_back({i,s[i]});
        }
    }
    sort(ur.begin(),ur.end(),cmp);
    for (int i = 0;i < ur.size() - 1;i ++)
    {
        ans += ur[i].second;
    }
    cout << ans;
    return 0;
}
