小红的图上加边题目链接

这个题本质上是记录每次操作(将元素并到另一个集合里),计算代价并取和
大体上是考察简单的并查集
剩余的一些细节看代码

代码如下:

#include <algorithm>
#include <iostream>
#include <cstring>
#include <vector>
#define int long long

using namespace std;

const int N = 100010;

int e[N],ne[N],h[N],idx;

int n,m,ans;

int ar[N],p[N],s[N];

vector<pair<int,int>> ur;

void add(int a,int b) {e[idx] = b,ne[idx] = h[a],h[a] = idx ++;}

int find(int x) {if (p[x] != x) p[x] = find(p[x]);return p[x];}

bool cmp(pair<int,int> a,pair<int,int> b) {return a.second > b.second;}

signed main()
{
    cin >> n >> m;
    //并查集操作
    for (int i =1 ;i<= n;i ++) p[i] = i,cin >> s[i];
    while (m --)
    {
        int a,b;
        cin >> a >> b;
        if (find(a) != find(b))
        {
            a = find(a),b = find(b);
            s[b] = max(s[b],s[a]);
            p[a] = b;
        }
    }
    for (int i = 1;i <= n;i ++)
    {
        if (p[i] == i)
        {
            ur.push_back({i,s[i]});//录入所有根节点
        }
    }
    sort(ur.begin(),ur.end(),cmp);
    for (int i = 0;i < ur.size() - 1;i ++)//根据根节点权值从小到大排序,进行取和即可.
    {
        ans += ur[i].second;
    }
    cout << ans;
    return 0;
}
