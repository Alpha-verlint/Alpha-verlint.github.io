氧气少年的水滴题目链接

大体思路: 这道题考察链表的操作,不过需要注意的是本题不需要进行麻烦的 去点.接点 操作,因为是从选定的一个点进行初次操作并进行后续判断,如果水滴满足>10条件,则按顺序每次枚举左(或右)侧的点即可,可以单独定义两个变量进行移位,不需要删除关系,也不需要判断关系.
剩余的一些细节和详细解答,看代码.

代码如下:

#include <iostream>
#include <cstring>

using namespace std;

const int N = 200010;

int ar[N];

void solve()
{
    int n,p;
    cin >> n >> p;
    for (int i = 1;i <= n;i ++) cin >> ar[i];
    ar[p] ++;
    int fl = 0,fr = 0;
    int i = p - 1,j = p + 1;
    if (ar[p] >= 10) fl ++,fr ++;
    int ans1 = 0,ans2 = 0;
    while (fl || fr)
    {
        if (i == 0)
        {
            ans1 += fl;
            fl = 0;
        }
        if (j == n + 1)
        {
            ans2 += fr;
            fr = 0;
        }
        if (i != 0)
        {
            int num = 10 - ar[i];
            if (fl >= num)//这里是大于等于,等于10也会炸裂.
            {
                fl -= num;
                fl ++,fr ++;//注意这里水滴炸裂后两侧盈余值都++
                i --;
            }
            else
            {
                ar[i] += fl;
                fl = 0;
            }
        }
        if (j != n + 1)
        {
            int num = 10 - ar[j];
            if (fr >= num)
            {
                fr -= num;
                fr ++,fl ++;
                j ++;
            }
            else
            {
                ar[j] += fr;
                fr = 0;
            }
        }
    }
    cout << ans1 << ' ' << ans2 << endl;
}

int main()
{
    int t;
    cin >> t;
    while (t --) solve();
    return 0;
}
